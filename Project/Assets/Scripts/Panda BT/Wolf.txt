tree("Root")
	sequence
		tree "Idle"
		tree "CheckSurroundings"

tree("Idle")
	race
		CheckStats
		Idle

		
tree("Roaming")
	race
		while not isRoaming
			sequence
				DefineEndPoint
				CalculatePath
		while not isFollowingPath
			while isRoaming
				ResetValues

tree("Moving")
	race
		while not isRoaming
			sequence
				CalculatePath
				ToggleFoundFood
		while not isFollowingPath
			while isRoaming
				sequence
					EatSheep
				
tree("CheckSurroundings")
	sequence
		fallback
			while not isRoaming
				sequence	
					ToggleSurroundings
					Wait 2.0
					ToggleSurroundings
					TargetSheep
					tree "Moving"
			while foundFood
				tree "Moving"
