tree("Root")
	tree "Sheep"

tree("Sheep")
	sequence
		while not hasTarget
			CalculatePosition
		while not isCaught
			sequence
				tree "NextStep"

tree("NextStep")
	repeat
		fallback
			sequence
				random(70, 10)
					sequence
						NextStep
						CheckPath
						Succeed
					sequence
						tree "Sleep"
						Succeed
			while not hasTarget
				sequence
					Wait 2.0
					Fail

tree("Sleep")
	sequence
		Sleep
		Wait 5.0
		WakeUp
