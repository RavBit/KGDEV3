tree("Root")
	tree "Sheep"

tree("Sheep")
	fallback
		while not hasTarget
			CalculatePosition
		while isCaught
				tree "Follow"
		while not isCaught
			sequence
				tree "NextStep"

tree("NextStep")
	repeat
		fallback
			while isHit
				sequence
					tree "Defend"
			sequence
				random(70, 10)
					sequence
						NextStep
						CheckPath
						Succeed
					sequence
						Succeed
			while not hasTarget
				sequence
					Fail

tree("Defend")
	sequence
		Defend
		Wait 4.0
		Continue
		Succeed

tree("Follow")
	fallback
		while atHerder
			tree "CarryWooloo"
		sequence
			Follow
			NextStep
			Succeed

tree("CarryWooloo")
	sequence
		AttachToHerder
