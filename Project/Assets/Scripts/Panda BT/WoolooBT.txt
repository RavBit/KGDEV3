tree("Root")
	tree "Sheep"

tree("Sheep")
	sequence
		while not hasTarget
			CalculatePosition
		while not isCaught
			sequence
				tree "NextStep"

tree("NextStep")
	repeat
		fallback
			while isHit
				sequence
					tree "Defend"
			sequence
				random(70, 10)
					sequence
						NextStep
						CheckPath
						Succeed
					sequence
						Succeed
			while not hasTarget
				sequence
					Fail

tree("Sleep")
	sequence
		Sleep
		Wait 5.0
		WakeUp

tree("Defend")
	sequence
		Defend
		Wait 4.0
		Continue
		Succeed
